A generic browser input that takes [`MediaStream`](https://developer.mozilla.org/en-US/docs/Web/API/MediaStream) object.
Stream like this can be generated by user, or captured from various browser APIs like camera, screen share, or WebRTC connection.

<Code title='streamInputExample.tsx' code={`
import Smelter from "@swmansion/smelter-web-wasm"

async function run() {
    const stream = await navigator.mediaDevices.getUserMedia({
        audio: true,
        video: true
    });
    const smelter = new Smelter();
    await smelter.init();
    await smelter.registerInput('example', { type: 'stream', stream })
}
void run()
`} lang='tsx' />
